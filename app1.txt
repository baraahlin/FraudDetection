class FeatureGroupingEnv(gym.Env):
    def __init__(self, X, y, num_tasks=3):
        super(FeatureGroupingEnv, self).__init__()
        self.X, self.y = X, y
        self.num_features = X.shape
        self.num_tasks = num_tasks
        self.current_feature_idx = 0
        self.observation_space = spaces.Box(low=0, high=1, shape=(4,), dtype=np.float32)
        self.action_space = spaces.Discrete(self.num_tasks)
        self.feature_assignments = np.full(self.num_features, -1)
        self.selection_history = np.zeros(self.num_features)

    def _get_observation(self):
        f_idx = self.current_feature_idx
        np.random.seed(f_idx)
        return np.array([np.random.uniform(0.5, 1.0), 0.0, 0.5, self.selection_history[f_idx]], dtype=np.float32)

    def step(self, action):
        self.feature_assignments[self.current_feature_idx] = action
        self.selection_history[self.current_feature_idx] += 0.1
        self.current_feature_idx += 1
        done = self.current_feature_idx >= self.num_features
        reward = self._calculate_reward() if done else 0.0
        obs = np.zeros(4, dtype=np.float32) if done else self._get_observation()
        return obs, reward, done, False, {}

    def _calculate_reward(self):
        total_gmean = 0.0
        active_tasks = 0
        for t in range(self.num_tasks):
            task_indices = np.where(self.feature_assignments == t)
            if len(task_indices) < 3:
                total_gmean += 0.5
            else:
                knn = KNeighborsClassifier(n_neighbors=3)
                knn.fit(self.X[:, task_indices], self.y)
                y_pred = knn.predict(self.X[:, task_indices])
                total_gmean += geometric_mean_score(self.y, y_pred)
            active_tasks += 1
        if active_tasks == 0: return -20.0
        return (total_gmean / active_tasks) * 2.0

    def reset(self, seed=None, options=None):
        super().reset(seed=seed)
        self.current_feature_idx = 0
        self.feature_assignments = np.random.randint(0, self.num_tasks, size=self.num_features)
        return self._get_observation(), {}

class MFGWfs_Scaled_CC:
    def __init__(self, X, y, rl_task_assignments, num_wolves=10, max_iter=15): # Reduced for web speed
        self.X, self.y = X, y
        self.total_features = X.shape
        self.task_assignments = rl_task_assignments
        self.num_tasks = 3 
        self.num_wolves = num_wolves
        self.max_iter = max_iter
        self.rmp = 0.3
        self.tasks_features, self.populations = {}, {}
        self.alpha_pos, self.alpha_score = {}, {}
        self.beta_pos, self.beta_score = {}, {}
        self.delta_pos, self.delta_score = {}, {}
        self.fitness_history = []

        for t in range(self.num_tasks):
            self.tasks_features[t] = np.where(self.task_assignments == t)
            num_features = len(self.tasks_features[t])
            self.populations[t] = np.random.randint(2, size=(self.num_wolves, num_features)).astype(float)
            self.alpha_pos[t], self.beta_pos[t], self.delta_pos[t] = np.zeros(num_features), np.zeros(num_features), np.zeros(num_features)
            self.alpha_score[t], self.beta_score[t], self.delta_score[t] = -np.inf, -np.inf, -np.inf

    def fitness_function(self, wolf_binary, task_id):
        active_indices = np.where(wolf_binary >= 0.5)
        selected_cols = list(self.tasks_features[task_id][active_indices])
        for other_task in range(self.num_tasks):
            if other_task != task_id and len(self.tasks_features[other_task]) > 0:
                best_mask = self.alpha_pos[other_task] >= 0.5
                if np.any(best_mask):
                    selected_cols.extend(self.tasks_features[other_task][best_mask])
        if len(selected_cols) == 0: return 0.0
        selected_cols = np.unique(selected_cols).astype(int)
        
        # Fast evaluation
        knn = KNeighborsClassifier(n_neighbors=3)
        knn.fit(self.X[:, selected_cols], self.y)
        y_pred = knn.predict(self.X[:, selected_cols])
        return geometric_mean_score(self.y, y_pred)

    def optimize(self):
        progress_bar = st.progress(0)
        for iteration in range(self.max_iter):
            a = 2.0 - iteration * (2.0 / self.max_iter)
            for t in range(self.num_tasks):
                num_features = len(self.tasks_features[t])
                if num_features == 0: continue
                for i in range(self.num_wolves):
                    score = self.fitness_function(self.populations[t][i], t)
                    if score > self.alpha_score[t]:
                        self.delta_score[t], self.delta_pos[t] = self.beta_score[t], copy.deepcopy(self.beta_pos[t])
                        self.beta_score[t], self.beta_pos[t] = self.alpha_score[t], copy.deepcopy(self.alpha_pos[t])
                        self.alpha_score[t], self.alpha_pos[t] = score, copy.deepcopy(self.populations[t][i])
                    elif score > self.beta_score[t]:
                        self.delta_score[t], self.delta_pos[t] = self.beta_score[t], copy.deepcopy(self.beta_pos[t])
                        self.beta_score[t], self.beta_pos[t] = score, copy.deepcopy(self.populations[t][i])
                    elif score > self.delta_score[t]:
                        self.delta_score[t], self.delta_pos[t] = score, copy.deepcopy(self.populations[t][i])

            # Update wolves
            for t in range(self.num_tasks):
                num_features = len(self.tasks_features[t])
                if num_features == 0: continue
                for i in range(self.num_wolves):
                    for j in range(num_features):
                        r1, r2 = np.random.rand(), np.random.rand()
                        A1, C1 = 2 * a * r1 - a, 2 * r2
                        D_alpha = abs(C1 * self.alpha_pos[t][j] - self.populations[t][i][j])
                        X1 = self.alpha_pos[t][j] - A1 * D_alpha
                        # Simplified update for web speed
                        new_pos = X1 
                        sigmoid = 1.0 / (1.0 + np.exp(-10 * (new_pos - 0.5)))
                        self.populations[t][i][j] = 1 if sigmoid > np.random.rand() else 0

            # Record best score for graphs
            best_iter_score = max([self.alpha_score[t] for t in range(self.num_tasks) if self.alpha_score[t] != -np.inf], default=0)
            self.fitness_history.append(best_iter_score)
            progress_bar.progress((iteration + 1) / self.max_iter)
            
        progress_bar.empty()
